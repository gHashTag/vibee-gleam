## Database Schema

```sql
-- Agents table
CREATE TABLE agents (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  type TEXT NOT NULL,  -- lead_generator, support, sales, community
  session_id TEXT NOT NULL,
  status TEXT DEFAULT 'stopped',  -- running, stopped, paused
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Agent configuration
CREATE TABLE agent_configs (
  agent_id TEXT PRIMARY KEY REFERENCES agents(id),
  auto_reply_enabled BOOLEAN DEFAULT true,
  digital_twin_enabled BOOLEAN DEFAULT false,
  cooldown_ms INTEGER DEFAULT 30000,
  confidence_threshold FLOAT DEFAULT 0.7,
  llm_model TEXT DEFAULT 'x-ai/grok-4.1-fast',
  forward_chat_id TEXT,
  config_json JSONB  -- Full config
);

-- Target chats per agent
CREATE TABLE agent_target_chats (
  agent_id TEXT REFERENCES agents(id),
  chat_id TEXT NOT NULL,
  chat_name TEXT,
  added_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (agent_id, chat_id)
);

-- Trigger words per agent
CREATE TABLE agent_triggers (
  agent_id TEXT REFERENCES agents(id),
  trigger_word TEXT NOT NULL,
  added_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (agent_id, trigger_word)
);

-- Agent statistics
CREATE TABLE agent_stats (
  agent_id TEXT REFERENCES agents(id),
  date DATE DEFAULT CURRENT_DATE,
  messages_processed INTEGER DEFAULT 0,
  messages_sent INTEGER DEFAULT 0,
  leads_generated INTEGER DEFAULT 0,
  PRIMARY KEY (agent_id, date)
);
```
